Scanner leer = new Scanner(System.in); double base, altura, area; System.out.println("Digite la longitud de la base del triangulo."); base = leer.nextDouble(); System.out.println("Digite la longitud de la altura del triangulo."); altura = leer.nextDouble(); area = (base * altura) / 2; System.out.println("El area del triangulo es " + area + ".");
Scanner leer = new Scanner(System.in); double cateto1, cateto2, hipotenusa; System.out.println("Digite la longitud del primer cateto."); cateto1 = leer.nextDouble(); System.out.println("Digite la longitud del primer cateto."); cateto2 = leer.nextDouble(); hipotenusa = Math.sqrt(Math.pow(cateto1, 2) + Math.pow(cateto2, 2)); System.out.println("La longitud de la hipotenusa es " + hipotenusa + ".");
Scanner leer = new Scanner(System.in); double radio, area, circunferencia; System.out.println("Digite el radio del circulo."); radio = leer.nextDouble(); area = Math.PI * Math.pow(radio, 2); circunferencia = 2 * Math.PI * radio; System.out.println("El area del circulo es " + area + "."); System.out.println("La circunferencia del circulo es " + circunferencia + ".");
Scanner leer = new Scanner(System.in); int ano1, ano2, mes1, mes2, distancia; System.out.println("Digite el año inicial."); ano1 = leer.nextInt(); System.out.println("Digite el mes inicial."); mes1 = leer.nextInt(); System.out.println("Digite el año final."); ano2 = leer.nextInt(); System.out.println("Digite el mes final."); mes2 = leer.nextInt(); distancia = ((ano2 - ano1) * 12) + (mes2 - mes1); System.out.println("Las fechas se encuentran a una distancia de " + distancia + " meses entre si.");
Scanner leer = new Scanner(System.in); double tanque, manguera, minutos, horas; tanque = 50 * 1000; System.out.println("Digite la capacidad de la manguera."); manguera = leer.nextDouble(); minutos = tanque / manguera; horas = minutos / 60; System.out.println("Tardaria " + horas + " en llenarse el tanque.");
